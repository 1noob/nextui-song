void get_next(char* p){
   memset(next,0,sizeof next);

       while(j && p[i]!=p[j])

   }
}

int KMP(char* w,char* s){
   int ans=0, j=0;
   // 对单词 w 进行预处理
   get_next(w);
   for(int i=0;i<ls;i++){
       while(j && s[i]!=w[j]) j=next[j];
       if(s[i]==w[j]) j++;
       if(j==lw) ans++;
   }
   return ans;
}`}
